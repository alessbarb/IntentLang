## 2) Crear un proyecto

```bash
# Minimal o con tests
intent init . --template minimal
# o
intent init ./hello-il --template tests -y
```

Estructura generada:

```text
ilconfig.json
src/hello.il
```

`ilconfig.json` (editable):

```json
{
  "compilerOptions": {
    "strict": true,
    "target": "ts",
    "outDir": "dist",
    "seedRng": "42",
    "seedClock": "0"
  },
  "include": ["src/**/*.il"]
}
```

## 3) Tu primer archivo `.il`


## 7) Conceptos clave del lenguaje

### 7.1 Brands

```intentlang
types
  type UserId = brand "UserId" string

func
  mkUser(id: UserId): string { "user#" + id }
```

### 7.2 Records y Unions

```intentlang
types
  type User = {
    id: UserId
    name: string
    email: string
  }

  type Payment = Card | Cash

  type Card = { last4: string }
  type Cash = { }
```

### 7.3 `Option` y `Result` (sin null)

```intentlang
func
  head(xs: string[]): Option<string> {
    match xs {
      [] -> None
      [x, .._] -> Some(x)
    }
  }

func
  parseIntStrict(s: string): Result<number, string> {
    let n = toInt(s)
    match isNaN(n) {
      true -> Err("not a number")
      false -> Ok(n)
    }
  }
```

### 7.4 `match` exhaustivo

```intentlang
types
  type Shape = Circle | Rect
  type Circle = { r: number }
  type Rect = { w: number, h: number }

func
  area(s: Shape): number {
    match s {
      Circle{ r } -> 3.14159 * r * r
      Rect{ w, h } -> w * h
    }
  }
```

### 7.5 `func` vs `effect` + `uses`

```intentlang
effect
  // Declaras capacidades que usar√°s (ej. "console")
  logHello(name: string): void
    uses console
  {
    console.log("Hello, " + name)
  }
```
